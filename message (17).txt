                function pollTaskResult(taskId, attempt) {
                    if (attempt > 60) {
                        showError('Request timeout');
                        return;
                    }
                    setTimeout(() => {
                        GM_xmlhttpRequest({
                            method: 'GET',
                            url: `${MEOBYPASS_API}/taskid/${taskId}`,
                            onload: function(response) {
                                try {
                                    const data = JSON.parse(response.responseText);
                                    if (data.status === 'success' && data.result) {
                                        const isUrl = data.result.startsWith('http://') || data.result.startsWith('https://');
                                        showSuccess(isUrl, data.result);
                                    } else if (data.status === 'pending' || data.status === 'processing') {
                                        pollTaskResult(taskId, attempt + 1);
                                    } else {
                                        showError(data.message || 'Bypass failed', data.your_ip || null);
                                    }
                                } catch (e) {
                                    pollTaskResult(taskId, attempt + 1);
                                }
                            },
                            onerror: function() {
                                pollTaskResult(taskId, attempt + 1);
                            }
                        });
                    }, 2000);
                }
            }
